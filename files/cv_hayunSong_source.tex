\cvsection{Research Papers}
\iflongcv
\begin{enumerate}[itemsep=0.42em]
\else
\begin{enumerate}[itemsep=0.08em]
\fi
  \paper{Song, Hayun (2024)}{High-dimensional Bayesian nonparanormal dynamic conditional model with multivariate volatility applications}
  \abstracttext{This paper develops a Bayesian method for estimating large conditional precision matrices within a DCC--GARCH framework. Wishart likelihoods, horseshoe priors, and a nonparanormal rank transformation allow sparse conditional dependence estimation even when the number of variables exceeds the number of observations. Monte Carlo experiments and applications to international stock indices and asset-pricing models demonstrate the method's empirical usefulness.}

  \item Lee, Junghyuk and Hayun Song (2023). \emph{Female labor force participation and gender role attitudes}. [Manuscript in preparation].

  \paper{Song, Hayun (2021)}{Bayesian dynamic factor augmented structure learning: cross-sectional dependence for residuals}
  \abstracttext{This paper proposes a Bayesian dynamic factor-augmented VAR that represents contemporaneous cross-sectional dependence as a graphical model. Factors are estimated through principal components and forward-filtering backward-sampling, while rescaled spike-and-slab priors address high-dimensional transition matrices. A fractional Bayes factor supports graphical model selection, with simulations and an application to U.S. house prices illustrating the approach.}

  \paper{Song, Hayun (2018)}{Individual heterogeneity in the returns to schooling: instrumental variable quantile regression}
  \abstracttext{Using instrumental variable quantile regression on twins data, this paper studies heterogeneity in returns to schooling while addressing ability and measurement-error biases. Estimated returns range from 9 to 15 percent across quantiles, and Wald-type location-shift tests provide evidence of individual heterogeneity. The analysis also examines ability bias, measurement error, linearity, and heterogeneous returns to other worker characteristics.}
\end{enumerate}
